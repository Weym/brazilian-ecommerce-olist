import streamlit as st
import pandas as pd
import plotly.graph_objects as go
from pathlib import Path

THRESHOLD_LOW = threshold * 0.6    # abaixo: verde
THRESHOLD_HIGH = threshold         # acima: vermelho


def build_gauge(prob: float) -> tuple:
    """
    Transforma probabilidade float 0-1 em gauge visual e acao recomendada.
    Retorna (fig, action_text, color_label).
    """
    pct = prob * 100  # exibir como 0-100%

    if prob < THRESHOLD_LOW:
        action = "Sem acao necessaria — pedido dentro do padrao"
        bar_color = "#2ecc71"
        risk_label = "Risco Baixo"
    elif prob < THRESHOLD_HIGH:
        action = "Monitorar — acompanhar prazo e rastrear entrega"
        bar_color = "#f39c12"
        risk_label = "Risco Medio"
    else:
        action = "Contato preventivo via WhatsApp antes da data estimada de entrega"
        bar_color = "#e74c3c"
        risk_label = "Risco Alto"

    fig = go.Figure(go.Indicator(
        mode="gauge+number",
        value=pct,
        number={"suffix": "%", "font": {"size": 40}},
        title={"text": f"Score de Risco Pre-Entrega<br><span style='font-size:0.7em'>{risk_label}</span>"},
        gauge={
            "axis": {"range": [0, 100], "tickwidth": 1, "tickcolor": "darkgray"},
            "bar": {"color": bar_color, "thickness": 0.25},
            "bgcolor": "white",
            "borderwidth": 2,
            "bordercolor": "gray",
            "steps": [
                {"range": [0, THRESHOLD_LOW * 100], "color": "#d5f5e3"},
                {"range": [THRESHOLD_LOW * 100, THRESHOLD_HIGH * 100], "color": "#fdebd0"},
                {"range": [THRESHOLD_HIGH * 100, 100], "color": "#fadbd8"},
            ],
            "threshold": {
                "line": {"color": "black", "width": 4},
                "thickness": 0.75,
                "value": pct,
            },
        },
    ))
    fig.update_layout(
        height=300,
        margin={"t": 60, "b": 0, "l": 40, "r": 40},
    )
    return fig, action


# --- UI ---
st.title("Preditor de Risco Pre-Entrega")
st.markdown(

            help="Frete cobrado no pedido"
        )
        payment_value = st.number_input(
            "Valor do Pedido (R$)",
            min_value=0.0, max_value=10000.0, value=100.0, step=1.0,
            help="Valor total pago pelo cliente"
        )
        estimated_days = st.number_input(
            "Prazo Estimado (dias)",
            min_value=1, max_value=90, value=10,
            help="Dias ate a data estimada de entrega"
        )

    with col2:
        categoria = st.selectbox(

            options=categories if categories else ["(Phase 3 pendente — sem categorias)"],

        uf_origem = st.selectbox(

            index=ufs.index("SP") if "SP" in ufs else 0,
            help="Estado onde o vendedor esta localizado"
        )
        uf_destino = st.selectbox(
            "UF de Destino (comprador)",
            options=ufs,
            index=ufs.index("RJ") if "RJ" in ufs else 0,
            help="Estado do comprador"
        )

    submitted = st.form_submit_button("Calcular Risco", type="primary", use_container_width=True)

if submitted:
    # CRITICO: chaves do dict devem bater EXATAMENTE com os nomes usados no treino da Phase 4
    input_data = {

            st.write(f"Threshold baixo (x0.6): {THRESHOLD_LOW:.4f}")
            st.dataframe(input_df)

    except Exception as e:
        st.error(f"Erro ao calcular risco: {e}")
        st.info(
            "Verifique se os nomes das colunas batem com PRE_DELIVERY_FEATURES de src/features.py. "

        )
